contentEditable keyboard shortcuts do not work in Firefox.


Default Behaviour

The contend within this <div> is editable

Select this text, and make it bold with Ctrl + B

Select this text, and make it italics with Ctrl + I

Select this text, and make it underlined with Ctrl + U


Desired Behaviour

The contend within this <div> is editable

Select this text, and make it bold with Ctrl + B

Select this text, and make it italics with Ctrl + I

Select this text, and make it underlined with Ctrl + U


JavaScript Solution

  
const handleKeydown = (e) => {
  const key = e.key;
  const ctrl = e.ctrlKey;
  const meta = e.metaKey;

  if (ctrl || meta) {
    switch (key) {
      case 'b':
        e.preventDefault();
        document.execCommand("bold");
        break;
      case 'i':
        e.preventDefault();
        document.execCommand("italic");
        break;
      case 'u':
        e.preventDefault();
        document.execCommand("underline");
        break;
    }
  }
};
